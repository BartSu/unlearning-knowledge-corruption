\begin{frame}{一句话贡献（初步）}
    \begin{alertblock}{}
        在当前设置下，LLM unlearning 的 ppl 副作用沿语义距离呈三层衰减：\\[0.2em]
        \hspace*{1em}\textbf{L1 forget 1.76× \;→\; L2 locality 1.29× \;→\; L3 spillover 1.16×} \\[0.2em]
        但 L3 \emph{并未归零}（\textbf{61.6\,\%} 跨簇样本 surprisal 上升）。
    \end{alertblock}

    \begin{itemize}
        \item 10 个 forget set 上 L1 差 \textbf{1.7$\times$}
              \;$\Rightarrow$\; 提示单 forget-set benchmark \emph{稳定性不足}。
        \item 用 \textbf{12 维 forget-set 内禀几何}、\emph{不跑 unlearn}，
              LOO Spearman $\rho\approx\textbf{0.62}$ 预测三层风险。
        \item L1 层 top-1 预测与真实排序吻合（$n=10$，单观察点）
              $\Rightarrow$ \emph{forget-set 审计} 的 proof-of-concept。
    \end{itemize}

    \vspace{0.3em}
    \footnotesize\itshape
    所有结论均基于：单 base model（Llama-3.1-8B-Instruct）、单 unlearner
    （GradAscent）、$n=10$ 个 forget set。
\end{frame}

\section{动机与定义}

\begin{frame}{动机：unlearning 贵，副作用还反直觉}
    \begin{columns}[T,onlytextwidth]
        \column{0.54\textwidth}
        \textbf{痛点}
        \begin{itemize}
            \item Fine-tune 8B + 跨评测 $\Rightarrow$ 数 GPU-hour / forget-set。
            \item 同算法在不同 forget set 上副作用差 \textbf{1.7×}。
            \item 现有 unlearning 论文几乎只用 \emph{一个} forget set 报数字。
        \end{itemize}

        \column{0.46\textwidth}
        \textbf{研究问题}
        \vspace{0.3em}
        \begin{block}{}
            给候选 forget set，\emph{不 fine-tune} 就预测它的
            L1 / L2 / L3 corruption profile。
        \end{block}
    \end{columns}

    \vspace{0.8em}
    \centering
    \small 把 unlearning 从\textbf{"每次都跑"} 改为\textbf{"先审计、再决定"}。
\end{frame}

\begin{frame}{三层 Corruption View}
    对 base 模型 $f$ 与 $\mathcal{D}_f$ 上 GradAscent 后的 $f_{\mathcal{D}_f}$：
    \[
        r(f, \mathcal{D}_f, x) \;=\;
        \frac{\mathrm{PPL}(f_{\mathcal{D}_f}, x)}{\mathrm{PPL}(f, x)}.
    \]

    \vspace{0.3em}
    \centering\small
    \begin{tabular}{@{}lll@{}}
        \toprule
        层 & $x$ 与 $\mathcal{D}_f$ 的关系 & 直观含义 \\
        \midrule
        \textbf{L1} forget     & $x$ 就是 $\mathcal{D}_f$ 的 train 分片   & 故意遗忘强度 \\
        \textbf{L2} locality   & $x$ 来自同簇 retain/test             & 局部附带损伤 \\
        \textbf{L3} spillover  & $x$ 来自\emph{另一簇}                 & 跨簇知识污染 \\
        \bottomrule
    \end{tabular}

\end{frame}

\section{第一幕 · 数据证实三层 View}

\begin{frame}{这一幕要验证的三个论断}
    若 §定义中的三层切分真抓住了 corruption 结构，数据应同时满足：
    \vspace{0.5em}
    \begin{description}
        \item[\alert{C1 单调性}] $r$ 随语义距离 L1 → L2 → L3 递减。\\
            否则分层只是任意分箱，\emph{不是有序结构}。
        \item[\alert{C2 非零溢出}] L3 几何均值 $>1$ 且非零率可观。\\
            否则跨簇污染只是噪声，框架退化为 locality 分析。
        \item[\alert{C3 forget-set 依赖}] 三层 profile 在 forget set 间显著不同。\\
            否则三层只是算法常数，\emph{无审计可言}（§第二幕失去动机）。
    \end{description}
    \vspace{0.4em}
    \centering\small
    接下来两帧分别给出 \textbf{C1 + C2} 与 \textbf{C3} 的证据。
\end{frame}

\begin{frame}{C1 + C2：层级 headline（全池样本）}
    \centering
    \begin{tabular}{@{}lrrrr@{}}
        \toprule
        Layer & $n$ & geo-mean $r$ & $>1.1\times$ & $>2\times$ \\
        \midrule
        L1 forget     &   500 & \textbf{1.762$\times$} & 95.6\,\% & 33.2\,\% \\
        L2 locality   & 1\,000 & \textbf{1.290$\times$} & 82.8\,\% &  3.4\,\% \\
        L3 spillover  & 4\,500 & \textbf{1.158$\times$} & \alert{61.6\,\%} &  0.0\,\% \\
        \bottomrule
    \end{tabular}

    \vspace{0.8em}
    \begin{itemize}
        \item \textbf{C1 成立}：1.762 → 1.290 → 1.158 严格单调；L2 保留 L1
              损伤 \textbf{38\,\%}，L3 保留 \textbf{21\,\%}。
        \item \textbf{C2 成立}：L3 geo-mean $=1.158\times$、\alert{61.6\,\%}
              跨簇样本 surprisal 真的上升 $\Rightarrow$ 溢出\emph{不可忽略}。
        \item 消歧参照：self (同簇 test) $1.291\times$ vs. cross $1.158\times$，
              locality multiplier 仅 \textbf{1.114$\times$} —— L2 只比 L3 高
              $\sim$11\,\%，\alert{局部性存在但比常见直觉弱}。
    \end{itemize}
\end{frame}

\begin{frame}{C3：三层 profile 随 forget set 而变（差 1.7$\times$）}
    \centering
    \scriptsize
    \begin{tabular}{@{}lrrrrr@{}}
        \toprule
        forget set & 簇 & L1 & L2 & L3\_out & \\
        \midrule
        triplet\_091 & star    & 1.42 & 1.12 & 1.07 & \footnotesize 最温和 \\
        triplet\_001 & game    & 1.44 & 1.17 & 1.07 & \\
        triplet\_011 & federer & 1.45 & 1.11 & 1.07 & \\
        triplet\_021 & jordan  & 1.71 & 1.31 & 1.20 & \\
        triplet\_081 & river   & 1.73 & 1.23 & 1.18 & \\
        triplet\_061 & war     & 1.81 & 1.21 & 1.18 & \\
        triplet\_051 & song    & 1.86 & 1.29 & 1.19 & \\
        triplet\_041 & league  & 1.99 & 1.42 & 1.20 & \\
        triplet\_031 & episode & 2.03 & 1.39 & 1.19 & \\
        \alert{triplet\_071} & \alert{storm}   & \alert{\textbf{2.44}} & \alert{\textbf{1.78}} & \alert{\textbf{1.25}} & \footnotesize 最严重 \\
        \bottomrule
    \end{tabular}

    \vspace{0.8em}
    \begin{itemize}\small
        \item \textbf{C3 成立}：L1 跨 forget set 差 \textbf{1.72$\times$}、
              L2 差 1.60$\times$、L3\_out 差 1.17$\times$ —— profile 是
              forget set 的函数，不是算法常数。
        \item 嵌入紧凑度（低 emb variance）与 L2 的 Spearman $\rho=0.50$
              ($n=10$，CI 较宽，提示性证据）—— 已暗示\emph{几何可预测
              corruption}（第二幕将正式化）。
        \item 方法学含义：\alert{仅用单 forget set 给 unlearner 打分会显著
              低估评分方差} $\Rightarrow$ suggests benchmark instability，
              而非"系统性误报"。
    \end{itemize}

    \vspace{0.4em}
    \centering\footnotesize
    \emph{小结}：C1 + C2 说明三层是\emph{有序且 L3 非平凡}的结构；C3 说明
    profile 因 forget set 而异 $\Rightarrow$ \textbf{预测 profile 是良定义
    问题}（$\to$ 第二幕）。
\end{frame}

\section{第二幕 · 与其每次 unlearn，不如审计}

\begin{frame}{审计问题}
    \begin{block}{Forget-Set Audit}
        给候选 $\mathcal{D}_f$ 与 base 模型，\emph{不跑 fine-tune}，输出
        L1 / L2 / L3 三个风险分数。
    \end{block}

    \vspace{0.6em}
    \textbf{审计特征（12 维，纯 forget-set 内禀几何）}
    \begin{itemize}
        \item 方差：\texttt{emb\_variance\_\{mean,max\}}
        \item 两两：\texttt{pairwise\_sim\_\{mean,std,q90\}}, \texttt{pairwise\_eucl\_mean}
        \item Centroid / norm：\texttt{centroid\_norm}, \texttt{emb\_norm\_\{mean,std\}}
        \item 集中度：\texttt{effective\_rank}, \texttt{isotropy}, \texttt{spread\_over\_centroid}
    \end{itemize}

    \vspace{0.4em}
    \footnotesize
    $\bullet$ 不看任何 evaluation text $\bullet$ 不跑任何 unlearn
    $\bullet$ 10 forget set 上 Leave-One-Out + Ridge。
\end{frame}

\begin{frame}{审计结果：回归 + 排序}
    \centering
    \textbf{回归（LOO over 10 forget sets）}\\[0.3em]
    \begin{tabular}{@{}lrrr@{}}
        \toprule
        审计目标 & $R^2$ & Spearman $\rho$ & Pearson $r$ \\
        \midrule
        L1 forget     & \textbf{+0.443} & +0.624 & +0.684 \\
        L2 locality   & \textbf{+0.410} & +0.624 & +0.658 \\
        L3 spillover  & +0.190          & \textbf{+0.612} & +0.512 \\
        \bottomrule
    \end{tabular}

    \vspace{1.0em}
    \textbf{排序（predicted rank vs. true rank）}\\[0.3em]
    \begin{tabular}{@{}lccc@{}}
        \toprule
        Layer & Spearman $\rho$ & Top-1 命中 & Top-3 重合 \\
        \midrule
        L1 forget     & +0.624 & \alert{\checkmark storm} & 2\,/\,3 \\
        L2 locality   & +0.624 & $\times$ & 2\,/\,3 \\
        L3 spillover  & +0.612 & $\times$ & 1\,/\,3 \\
        \bottomrule
    \end{tabular}

    \vspace{0.8em}
    \footnotesize
    L1 层 top-1 与真实排序一致（triplet\_071 storm，\emph{单观察点}，$n=10$）。
    初步证据\emph{支持}审计器作为 \alert{排序 / 筛选工具} 的可行性；是否对更大
    $n$、其他 unlearner、其他 base model 稳健，\emph{尚需验证}。
\end{frame}

\begin{frame}{Per-sample 对照：几何 $\gg$ 文本表面}
    \centering
    \begin{tabular}{@{}lrr@{}}
        \toprule
        设定 & 特征数 & LOGO $R^2$ \\
        \midrule
        旧：261 维表格（含 per-text 表面统计）   & 261 & +0.142 \\
        \textbf{新：16 维纯几何（forget + target↔forget）} & \textbf{16} & \textbf{+0.362} (GBM) \\
        仅限 L3 跨簇样本（纯 spillover）         & 16  & +0.247 (RF) \\
        \bottomrule
    \end{tabular}

    \vspace{1.0em}
    \begin{block}{解读（谨慎）}
        在当前设置下，几何特征以约 \alert{6\,\%} 的特征数带来 \alert{2.5$\times$}
        的 $R^2$。这\emph{支持} "corruption 与 forget-set 几何比与文本表面
        统计更相关" 的假设；在单 base model、单 unlearner、$n=10$ 下，尚不
        足以断言 "corruption 是几何现象而非文本难度现象"。
    \end{block}
\end{frame}

\begin{frame}{诚实负结果：Naive retain-coverage 不 work}
    把 forget set 的 p95 内部 euclidean 当"高风险半径"，测 retain 落入比例：

    \vspace{0.5em}
    \begin{itemize}
        \item 9\,/\,10 个 forget set 的 mean coverage $\approx 1.0$ —— 半径太大，
              \emph{区分度归零}。
        \item 与真实 L3 spillover 的相关：$\rho = \alert{-0.42}$（方向反了）。
    \end{itemize}

    \vspace{0.4em}
    \textbf{原因}：HDBSCAN 簇在 384 维原空间里互相重叠，p95 半径 > 簇间距。

    \vspace{0.6em}
    \textbf{改进方向}
    \begin{enumerate}
        \item 到 centroid 的\emph{带符号投影}，单尾切片而非球覆盖；
        \item HDBSCAN 原生 mutual-reachability 距离；
        \item 直接把 L3 predictor 当 calibrated per-text risk score。
    \end{enumerate}
\end{frame}

\section{讨论}

\begin{frame}{Related Work（占位，待程序化补全）}
    \begin{description}
        \item[Data-first 鲁棒性预测] Dang et al.\ (ACL 2024): 13 维训练集特征
            $\to$ 对抗 ASR。我们把方法论迁到 \emph{生成式 unlearning 的
            corruption profile 审计}。
        \item[LLM unlearning] GradAscent / NPO / GradDiff / TOFU 由
            \texttt{open-unlearning} 提供；我们\emph{固定 unlearner}，分析
            其 data-conditioned 副作用。
        \item[Influence / editing locality] Model-centric 的局部性已被广研；
            我们用 10$\times$10 cross-ppl 矩阵把它变成可度量的三层风险。
    \end{description}
\end{frame}

\begin{frame}{Limitations}
    \begin{itemize}
        \item \textbf{$n=10$ clusters} —— 审计 $\rho=0.62$ 置信区间较宽；
              100 triplet 全跑可升级到 $n=100$。
        \item \textbf{单 unlearner} —— L1/L2/L3 对 NPO / GradDiff 的稳健性
              待测。
        \item \textbf{单 base model} —— Llama-3.1-8B-Instruct；其他规模 / 家族
              未测。
        \item \textbf{ppl $\neq$ fact} —— L3 的 61.6\,\% 样本是否对应 QA 准确率
              下降？\texttt{2.extract-qa/} 管线已就位。
        \item \textbf{Naive coverage 失败} —— §5.6 需要更细的几何风险度量。
    \end{itemize}
\end{frame}

\begin{frame}{Future Work}
    \begin{enumerate}
        \item \textbf{Scale-up}：100 triplet 全跑 unlearn $\Rightarrow$ $n=100$。
        \item \textbf{跨 unlearner}：加 NPO / GradDiff，审计器须稳健。
        \item \textbf{更好的 coverage}：方向投影 / mutual-reachability。
        \item \textbf{Audit-driven forget 选择}：在 retain 约束下挑
              predicted L3 最小的 $\mathcal{D}_f$，真跑 unlearn 验证。
        \item \textbf{QA 层面}：target 换成 QA accuracy。
    \end{enumerate}
\end{frame}

\begin{frame}{论文定位}
    \begin{itemize}
        \item \textbf{Title（工作名）}：\emph{A Three-Layer View of Knowledge
              Corruption in LLM Unlearning, and Why You Should Audit Before
              You Forget.}
        \item \textbf{Venue}：COLM 2025 / ACL short / EMNLP findings ——
              empirical + analysis + methodological 三位一体。
        \item \textbf{卖点（以当前证据强度措辞）}：
              (1) 三层 view $+$ 61.6\,\% 的 L3 spillover 提示\emph{spillover
                  不可忽略}；
              (2) 把 unlearning benchmarking 重构为 \emph{forget-set audit}；
              (3) 仅 forget-set 几何即可做排序级风险预测（LOO $\rho\approx 0.62$，
                  L1 top-1 匹配为单观察点）。
        \item \textbf{升档}：补到 $n=100$ + 第二个 unlearner
              $\Rightarrow$ NeurIPS / ICML。
    \end{itemize}
\end{frame}
