\section{问题与 setup}

\begin{frame}{研究问题：unlearn 会伤到无关数据}
    \vspace{0.3em}
    \textbf{背景} \;
    版权 / 隐私 / 有害内容 $\Rightarrow$ 法律 \& 伦理要求从 LLM 里
    \emph{定向删除}特定训练数据 $\Rightarrow$ \textbf{LLM unlearning}。

    \vspace{0.9em}
    \textbf{但 unlearning 不是 surgical}：
    \begin{itemize}
        \setlength\itemsep{0.3em}
        \item 让模型忘掉 forget set $\mathcal{D}_f$，会\textbf{同时损伤}与 $\mathcal{D}_f$ 无关的知识
        \item 我们称之为 \alert{knowledge corruption}，其半径与后果尚未被系统刻画
    \end{itemize}

    \vspace{0.8em}
    \textbf{Benchmark 痛点}：
    \begin{itemize}
        \setlength\itemsep{0.3em}
        \item 真跑一次 unlearn + $N\!\times\!N$ cross-eval $\approx$ \textbf{数 GPU·hour}
        \item 扫 100 个 candidate forget set $\Rightarrow$ 数百 GPU·hour，\emph{不 scale}
    \end{itemize}

    \vspace{0.8em}
    \begin{center}
        \Large
        \alert{问题：怎么\textbf{便宜、系统}地评估一个 forget set 会造成的 corruption？}
    \end{center}
\end{frame}

\begin{frame}{实验 setup：triplet schema + cross-eval 矩阵}
    \vspace{0.3em}
    \textbf{数据}：WikiText-103 $\to$ HDBSCAN 聚成 \textbf{10 个语义簇}\footnotesize{（Federer / Harry Potter / tennis-match / …）}\small。

    \vspace{0.6em}
    \textbf{Triplet schema}（每簇 10 个 triplet，共 \textbf{100}）：
    \begin{center}
    \small
    \begin{tabular}{@{}lll@{}}
        \toprule
        split & 50 条文本 & 下游用途 \\
        \midrule
        \texttt{train}      & \textbf{forget set} $\mathcal{D}_f$   & 喂给 unlearn（GradAscent） \\
        \texttt{validation} & retain 邻居（同簇）     & retain 侧 loss + L2 probe \\
        \texttt{test}       & probe 样本（同簇）      & L2 / L3 评测 \\
        \bottomrule
    \end{tabular}
    \end{center}
    三份\emph{同簇但互不相交} $\Rightarrow$ 天然 measure 三种语义距离。

    \vspace{0.6em}
    \textbf{评测}：每个 ckpt 在\emph{全部}其它 triplet 上评 PPL $\Rightarrow$ $N \!\times\! N$ cross-matrix
    $\Rightarrow$ 对角 / 非对角 $\Rightarrow$ L1 / L2 / L3。
\end{frame}

\section{第一幕 · 三层真的存在}

\begin{frame}{这一幕要说服你三件事}
    \vspace{0.8em}
    \begin{description}
        \setlength\itemsep{0.8em}
        \item[\alert{C1}] 三层\textbf{严格递减}（L1 > L2 > L3），不是乱分箱。
        \item[\alert{C2}] L3 \textbf{不是 0}：无关样本也真被波及。
        \item[\alert{C3}] profile \textbf{随 forget set 变化}，所以\emph{值得审计}。
    \end{description}
\end{frame}

\begin{frame}{C1 + C2：headline 数字}
    \vspace{0.2em}
    \centering
    \small
    $r = \dfrac{\text{PPL}_{\text{unlearned}}}{\text{PPL}_{\text{base}}}$ \;（PPL 被抬高几倍）\quad
    geo-mean $r = \exp(\overline{\log r})$

    \vspace{0.6em}
    \begin{tabular}{@{}lccc@{}}
        \toprule
        & L1 forget & L2 locality & L3 spillover \\
        \midrule
        geo-mean $r$           & \textbf{2.13×} & \textbf{1.49×} & \textbf{1.28×} \\
        被抬高（$>1.1×$）占比    & 99.0\,\%       & 91.9\,\%       & \alert{80.3\,\%} \\
        \bottomrule
    \end{tabular}

    \vspace{1.2em}
    \begin{center}
        \large
        \textbf{C1 成立}：2.13 $>$ 1.49 $>$ 1.28\\[0.4em]
        \textbf{C2 成立}：L3 仍有 \alert{80.3\,\%} 样本被抬高
    \end{center}
\end{frame}

\section{第二幕 · 先审计，再决定}

\begin{frame}{审计问题：输入是什么、输出是什么}

    \begin{center}
        \begin{tikzpicture}[font=\small, node distance=10mm]

            \node[draw, rounded corners, fill=mLightBrown!25, right=16mm of box, minimum width=32mm, minimum height=12mm, align=center] (out)

    \centering
    \footnotesize
    \begin{tabular}{@{}lll@{}}

        \midrule

        \midrule
        \textbf{位置 / 规模} & \texttt{centroid\_norm}
            & forget set 在空间里\\
        & \texttt{emb\_norm\_mean / std} & 离原点多远？ \\
        \midrule
        \textbf{形状 / 集中度} & \texttt{effective\_rank}
            & 这团点云是"一根线"\\
        & \texttt{isotropy} & 还是"一个球"？ \\
        & \texttt{spread\_over\_centroid} & \\
        \bottomrule
    \end{tabular}

    \vspace{0.8em}
    \begin{center}
        \normalsize
        \textbf{全部只看 forget set 自己的 50 条 embedding}\\
        不看任何 evaluation 文本、不跑 unlearn。
    \end{center}
\end{frame}

\begin{frame}{审计器的定位：预警器，不是替代品}

    \end{center}

\begin{frame}{Future Work}
    \vspace{0.3em}
    \begin{center}
        \large

    \end{center}

    \vspace{0.6em}
    \begin{enumerate}
        \setlength\itemsep{0.5em}
        \item $n: 50 \to 100$（100 triplet 已全部 unlearn，待扩 cross-eval）
        \item 加 NPO / GradDiff 验证审计器稳健
        \item PPL $\to$ QA accuracy（\texttt{2.extract-qa/} 已就位）
        \item \emph{audit-driven} forget 选择：挑预测 L3 最低的 $\mathcal{D}_f$
    \end{enumerate}
\end{frame}

\begin{frame}{论文定位}
    \begin{center}
        \large
        \emph{From Three-Layer Corruption to Forget-Set Audit\\ in LLM Unlearning.}
    \end{center}

    \vspace{0.8em}
    \textbf{三个卖点}
    \begin{enumerate}
        \item 三层 view + L3 80.3\,\% $\Rightarrow$ spillover \emph{不可忽略}
        \item benchmarking $\to$ \textbf{forget-set audit} 视角转换
        \item 仅几何就能排序预测（L1/L2 $\rho \approx 0.49\text{--}0.54$，CI 显著排除 0）
    \end{enumerate}

    \vspace{0.6em}
    \begin{center}
        \small
        \textbf{升档}：$n$=50 $\to$ 100 + 第二个 unlearner $\Rightarrow$ NeurIPS / ICML
    \end{center}
