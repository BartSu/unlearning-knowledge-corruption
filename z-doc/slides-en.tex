\section{Act I $\cdot$ The three layers are real}

\begin{frame}{Three things this act will show}
    \vspace{0.8em}
    \begin{description}
        \setlength\itemsep{0.8em}
        \item[\alert{C1}] The three layers are \textbf{strictly decreasing} (L1 $>$ L2 $>$ L3), not arbitrary bins.
        \item[\alert{C2}] L3 is \textbf{not zero} --- innocent samples really are hit.
        \item[\alert{C3}] The profile \textbf{varies with the forget set}, so \emph{auditing is worthwhile}.
    \end{description}
\end{frame}

\begin{frame}{C1 + C2: headline numbers}
    \vspace{0.2em}
    \centering
    \small
    $r = \dfrac{\text{PPL}_{\text{unlearned}}}{\text{PPL}_{\text{base}}}$ \;(how many times PPL is raised)\quad
    geo-mean $r = \exp(\overline{\log r})$

    \vspace{0.6em}
    \begin{tabular}{@{}lccc@{}}
        \toprule
        & L1 forget & L2 locality & L3 spillover \\
        \midrule
        geo-mean $r$              & \textbf{1.76$\times$} & \textbf{1.29$\times$} & \textbf{1.16$\times$} \\
        fraction raised ($>1.1\times$) & 95.6\,\%       & 82.8\,\%       & \alert{61.6\,\%} \\
        \bottomrule
    \end{tabular}

    \vspace{1.2em}
    \begin{center}
        \large
        \textbf{C1 holds}: 1.76 $>$ 1.29 $>$ 1.16\\[0.4em]
        \textbf{C2 holds}: L3 still raises \alert{61.6\,\%} of samples
    \end{center}
\end{frame}

\begin{frame}{C3: how much does side-effect vary across forget sets?}

\section{Act II $\cdot$ Audit first, then decide}

\begin{frame}{The audit problem: input and output}
    \vspace{0.4em}
    \begin{center}
        \begin{tikzpicture}[font=\small, node distance=10mm]
            \node[draw, rounded corners, fill=mDarkTeal!12, minimum width=32mm, minimum height=12mm, align=center] (in)
                {\textbf{Input}\\ 12-dim geometry\\ of the forget set};
            \node[draw, rounded corners, fill=mLightGreen!25, right=16mm of in, minimum width=28mm, minimum height=12mm, align=center] (box) {\textbf{Auditor}\\ Ridge};
            \node[draw, rounded corners, fill=mLightBrown!25, right=16mm of box, minimum width=32mm, minimum height=12mm, align=center] (out)
                {\textbf{Output}\\ L1 / L2 / L3\\ risk scores};
            \draw[-Latex, thick] (in) -- (box);
            \draw[-Latex, thick] (box) -- (out);
        \end{tikzpicture}
    \end{center}

    \vspace{0.8em}
    \begin{center}
        \large
        \alert{No evaluation text $\cdot$ no unlearn run}
    \end{center}

    \vspace{0.6em}
    \footnotesize\centering
    12 dims = variance / pairwise similarity / centroid / concentration (effective rank, isotropy).\\
    Leave-One-Out over 10 forget sets.
\end{frame}

\begin{frame}{What are the 12 geometric features?}
    \centering
    \footnotesize
    \begin{tabular}{@{}lll@{}}
        \toprule
        Group & Features (12 total) & Question answered \\
        \midrule
        \textbf{Spread}   & \texttt{emb\_variance\_mean / max}
            & How widely are the 50\\
                          & \texttt{pairwise\_eucl\_mean} & texts spread in embedding space? \\
        \midrule
        \textbf{Similarity} & \texttt{pairwise\_sim\_mean / std / q90}
            & How similar are the texts\\
                            & & to each other? \\
        \midrule
        \textbf{Location / scale} & \texttt{centroid\_norm}
            & How far from the origin\\
                                  & \texttt{emb\_norm\_mean / std} & does the forget set sit? \\
        \midrule
        \textbf{Shape / concentration} & \texttt{effective\_rank}
            & Is the cloud ``a line''\\
                                       & \texttt{isotropy} & or ``a ball''? \\
                                       & \texttt{spread\_over\_centroid} & \\
        \bottomrule
    \end{tabular}

    \vspace{0.8em}
    \begin{center}
        \normalsize
        \textbf{All computed from the 50 embeddings of the forget set itself ---}\\
        no evaluation text, no unlearn run.
    \end{center}
\end{frame}

\begin{frame}{Audit results}
    \centering
    \begin{tabular}{@{}lccc@{}}
        \toprule
        & L1 forget & L2 locality & L3 spillover \\
        \midrule
        $R^2$                    & \textbf{+0.44} & \textbf{+0.41} & +0.19 \\
        rank Spearman $\rho$     & +0.62 & +0.62 & +0.61 \\
        95\% CI ($n_\text{boot}=10$k) & [$-0.22$, $+0.93$] & [$+0.05$, $+0.90$] & [$-0.01$, $+0.96$] \\
        Top-1 hit                & \alert{\checkmark \, storm} & $\times$ & $\times$ \\
        \bottomrule
    \end{tabular}

    \vspace{0.9em}
    \begin{center}
        \Large
        $\rho \approx \textbf{0.62}$, but at $n=10$ the \alert{CI is wide}
    \end{center}
    \vspace{0.2em}
    \begin{center}
        \normalsize
        L1 / L3 lower bounds cross 0 $\Rightarrow$ \textbf{rank-level screening signal only}; $n=100$ needed to tighten.
    \end{center}

    \vspace{0.3em}
    \footnotesize\centering\itshape
    Percentile bootstrap over 10 LOO pairs; motivates the Stage-2 unlearn scale-up.
\end{frame}

\begin{frame}{Positioning: a warning tool, not a replacement}
    \centering
    \small
    \begin{tabular}{@{}lll@{}}
        \toprule
         & \textbf{Can} & \textbf{Cannot} \\
        \midrule
        \textbf{Rank}    & tell who is worse ($\rho \approx 0.62$) & absolute value \\
        \textbf{Screen}  & red / yellow / green warnings           & score an unlearner \\
        \textbf{Save compute} & 1 s vs. GPU-hours                  & replace the final unlearn \\
        \bottomrule
    \end{tabular}

    \vspace{1.0em}
    \begin{center}
        \Large
        \alert{Auditor = cheap coarse-screening warner}
    \end{center}

    \vspace{0.4em}
    \begin{center}
        \normalsize
        Usage: 100 candidates $\to$ audit rank $\to$ run the top-$k$ for real.\\
        Final verdict still comes from the actual unlearn run.
    \end{center}
\end{frame}

\begin{frame}{Geometry vs. surface text}
    \centering
    \begin{tabular}{@{}lcc@{}}
        \toprule
        Features & dim & $R^2$ \\
        \midrule
        surface text statistics    & 261         & +0.14 \\
        \textbf{pure geometry}     & \textbf{16} & \textbf{+0.36} \\
        \bottomrule
    \end{tabular}

    \vspace{1.2em}
    \begin{center}
        \Large
        \alert{16$\times$} fewer features, \alert{2.5$\times$} higher $R^2$
    \end{center}

    \vspace{0.4em}
    \begin{center}
        \normalsize
        Corruption looks more like a \emph{geometric} phenomenon than a text-difficulty one.
    \end{center}
\end{frame}

\begin{frame}{Honest negative result: ball-coverage doesn't work}
    \vspace{0.5em}
    \begin{center}
        \large
        Tried: circle retain samples by a radius from the forget set.
    \end{center}

    \vspace{0.8em}
    \begin{center}
        \large
        Got: correlation with true L3 is $\rho = \alert{-0.42}$ (\textbf{wrong sign}).
    \end{center}

    \vspace{0.8em}
    \begin{center}
        \normalsize
        Why: clusters overlap $\Rightarrow$ the ball covers almost everything.\\[0.3em]
        Fix: signed projection / mutual-reachability distance.
    \end{center}
\end{frame}

\begin{frame}{Limitations}
    \vspace{0.3em}
    \begin{center}
        \large
        Four ``singles''
    \end{center}

    \vspace{0.6em}
    \begin{itemize}
        \setlength\itemsep{0.6em}
        \item single base model (Llama-3.1-8B-Instruct)
        \item single unlearner (GradAscent)
        \item single $n = 10$ forget sets (wide CI)
        \item single metric: PPL ($\neq$ QA accuracy)
    \end{itemize}
\end{frame}

\begin{frame}{Future Work}
    \vspace{0.3em}
    \begin{center}

        \setlength\itemsep{0.5em}
        \item $n: 10 \to 100$ (100 triplets already generated)
        \item add NPO / GradDiff to check auditor robustness
        \item PPL $\to$ QA accuracy (\texttt{2.extract-qa/} ready)
        \item \emph{audit-driven} forget-set selection: pick $\mathcal{D}_f$ with lowest predicted L3
    \end{enumerate}
\end{frame}
